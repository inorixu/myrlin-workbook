getSession(id) { return this._state.sessions[id] || null; }

getWorkspaceSessions(workspaceId) {
  const ws = this.getWorkspace(workspaceId);
  if (!ws) return [];
  return ws.sessions.map(sid => this._state.sessions[sid]).filter(Boolean);
}

getAllSessionsList() {
  return Object.values(this._state.sessions).sort((a, b) =>
    new Date(b.lastActive || b.createdAt) - new Date(a.lastActive || a.createdAt)
  );
}

  <action>
1. In store.js, add a new method `getPaginatedSessions(options)` that accepts:
   - `limit` (number, default 50, max 100, min 1)
   - `offset` (number, default 0, min 0)
   - `status` (string: 'running', 'stopped', 'error', 'idle', 'all'; default 'all')
   - `sort` (string: 'lastActive', 'name', 'created'; default 'lastActive')
   - `order` (string: 'asc', 'desc'; default 'desc')
   - `search` (string, optional: case-insensitive substring match on name AND topic)
   - `workspaceId` (string, optional: filter to sessions in that workspace)

   Implementation:
   a. Start with `Object.values(this._state.sessions)`
   b. Filter by workspaceId if provided (check session.workspaceId matches)
   c. Filter by status if not 'all' (check session.status matches)
   d. Filter by search if provided (case-insensitive includes on session.name or session.topic)
   e. Sort by the sort field: 'lastActive' uses lastActive or createdAt fallback, 'name' is case-insensitive string compare, 'created' uses createdAt
   f. Apply order (asc/desc)
   g. Calculate total count BEFORE slicing
   h. Slice by offset and limit
   i. Return `{ sessions, total, limit, offset, hasMore: (offset + sessions.length) < total }`

2. In server.js, update the GET /api/sessions handler:
   - Keep backward compatibility: if `mode` param is present, use the existing switch/case logic (workspace, recent, all) and return `{ sessions }` as before
   - If ANY pagination param is present (limit, offset, status, sort, order, search, workspaceId WITHOUT mode), use the new `store.getPaginatedSessions()` and return `{ sessions, pagination: { total, limit, offset, hasMore } }`
   - If neither mode nor pagination params are present, default to the new paginated path with default limit=50 to maintain backward compat (existing callers get `{ sessions }` shape OR the paginated shape; since they only read `.sessions`, either works)
   - Actually, for full backward compat: when no pagination-specific params are provided (no limit, no offset, no status filter, no sort, no search), return the old `{ sessions }` format. When ANY pagination param is present, return `{ sessions, pagination }`.

   Detection logic:
   ```javascript
   const hasPaginationParams = req.query.limit || req.query.offset ||
     req.query.status || req.query.sort || req.query.order ||
     req.query.search || (req.query.workspaceId && !req.query.mode);
   ```

   Clamp limit: `Math.min(100, Math.max(1, parseInt(limit, 10) || 50))`
   Clamp offset: `Math.max(0, parseInt(offset, 10) || 0)`

3. Add JSDoc comments on the new store method and on the updated route handler.
