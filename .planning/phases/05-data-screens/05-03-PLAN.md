```
GET    /api/worktree-tasks                         -> { tasks: WorktreeTask[] }

PUT    /api/worktree-tasks/:id                     -> { task: WorktreeTask }
DELETE /api/worktree-tasks/:id                     -> { success: boolean }
POST   /api/worktree-tasks/:id/merge               -> { success, message }
POST   /api/worktree-tasks/:id/push                -> { success, message }
POST   /api/worktree-tasks/:id/reject              -> { success, message }
POST   /api/worktree-tasks/:id/pr                  -> { pr: { url, number, title } }
GET    /api/worktree-tasks/:id/pr                  -> { pr: { url, number, title, state, ... } | null }
POST   /api/worktree-tasks/:id/pr/generate-description -> { title, body }
GET    /api/worktree-tasks/:id/changes             -> { files: string[], stats: {...} }

POST /api/worktree-tasks body:
{
  workspaceId: string;
  repoDir: string;
  branch: string;
  description: string;
  baseBranch?: string;
  featureId?: string;
  model?: string;
  tags?: string[];
  startNow?: boolean;  // false = backlog mode
  prompt?: string;
  flags?: string[];
}

{
  id: string;

  baseBranch: string;

  status: 'backlog' | 'planning' | 'running' | 'review' | 'done';
  blockers: string[];

  updatedAt: string;
  branchAhead: number;
  changedFiles: number;
}
```

```typescript
  updatedAt: string;

}

interface CreateTaskInput {
  workspaceId: string;
  repoDir: string;
  branch: string;
  description: string;
  baseBranch?: string;
  featureId?: string;
  model?: string;
  tags?: string[];
  startNow?: boolean;
  prompt?: string;
  flags?: string[];
}

interface TaskPR {
  url: string;
  number: number;
  title: string;

}
```
