```typescript
// Response from /api/cost/dashboard
interface CostDashboardResponse {
  summary: {
    totalCost: number;           // Total cost in USD (all time)
    totalTokens: {
      input: number;
      output: number;
      cacheWrite: number;
      cacheRead: number;
    };
    periodCost: number;          // Cost within selected period
    periodLabel: string;         // e.g. "Last 7 days"
    messageCount: number;
    avgCostPerMessage: number;
    cacheSavings: number;        // Savings from cache reads in USD
  };
  timeline: Array<{
    date: string;     // "YYYY-MM-DD"
    cost: number;
    tokens: number;
    messages: number;
  }>;
  byModel: Array<{
    model: string;
    cost: number;
    tokens: number;
    pct: number;      // percentage of total
  }>;
  byWorkspace: Array<{
    id: string;
    name: string;
    cost: number;
    sessionCount: number;
    pct: number;
  }>;
  sessions: Array<{
    id: string;
    name: string;
    workspaceId: string;
    workspaceName: string;
    cost: number;
    messageCount: number;
    model: string;
    lastActive: string | null;
    firstMessage: string | null;
  }>;
}
```

<!-- Existing hook pattern from useSessions.ts -->
```typescript
export function useSessions(mode, params) {
  const client = useAPIClient();
  return useQuery({
    queryKey: ['sessions', mode, params],
    queryFn: () => client!.getSessions(mode, params),
    enabled: !!client,
    staleTime: 5000,
  });
}
```
